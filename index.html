Code Example 2: My project 3

This code snippet shows how to solve a simple differential equation using Euler's method.

# Python code for Euler's Method
def euler_method(dy_dt, y0, t0, t_end, h):
    t = t0
    y = y0
    while t < t_end:
        y += dy_dt(t, y) * h
        t += h
    return y

# Example differential equation: dy/dt = -y
def dy_dt(t, y):
    return -y

result = euler_method(dy_dt, 1, 0, 5, 0.1)
print(f"Result after 5 seconds: {result}")
